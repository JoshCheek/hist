```bash
ruby -r date -e '
(-30*10).step(to: 0, by: 0.01) do |offset|
  date = (Date.today + offset).strftime "%FT%T"
  File.write "date", date
  ENV["GIT_COMMITTER_DATE"] = date
  ENV["GIT_AUTHOR_DATE"]    = date
  system "git", "add", "date"
  system "git", "commit", "-m", date
end'
```
